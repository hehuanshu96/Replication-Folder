% Link to ToC from section

\newcommand{\maketoc}{
	\hypertarget{toc}{}
	\newpage
	\tableofcontents
	\vspace{2.5\bigskipamount} }

	}

	\end{itemize}
	\\ \hline
	\end{tabular} \\
	}
